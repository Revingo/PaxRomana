res_publica_on_start = {
	effect = {
		every_ruler = {
			limit = { 
				has_government = res_publica_government
				#is_indipendent = yes
				this = top_liege
			}

			if = {
				limit = {
					NOT = {
						any_vassal = {
							has_character_flag = co-consul_flag
						}
					}
				}

				set_respublica_first_time = yes
			}
		}
	}
}
