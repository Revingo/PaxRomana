res_publica_on_start = {
	effect = {
		every_ruler = {
			limit = { has_government = res_publica_government }

			if = {
				limit = {
					NOT = {
						any_vassal = {
							has_character_flag = co-consul_flag
						}
					}
				}

				set_co-consul_first_time = yes
			}

			create_alliance = this.var:co-consul
		}
	}
}
