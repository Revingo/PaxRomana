on_game_start_dynastic_setup = {
	effect = {
		every_ruler = {
			if = {
				limit = { this = {has_government = dynastic_government} }

				save_scope_as = current_d_ruler

				set_variable = {
					name = army_h_amount
					value = 50
				}
				set_variable = {
					name = nobles_h_amount
					value = 50
				}
				set_variable = {
					name = citizens_h_amount
					value = 50
				}

				random_vassal = {
					scope:current_d_ruler = {
						set_variable = {
							name = army_leader
							value = prev
						}

						set_variable = {
							name = nobles_leader
							value = prev
						}

						set_variable = {
							name = citizens_leader
							value = prev
						}
						
					}
				}
				every_vassal = {
		
					if = {
						limit = { scope:current_d_ruler.var:army_leader.martial < this.martial }

						scope:current_d_ruler = {
							remove_variable = army_leader
							set_variable = {
								name = army_leader
								value = prev
							}
						}
					}

					if = {
						limit = { scope:current_d_ruler.var:nobles_leader.stewardship < this.stewardship }

						scope:current_d_ruler = {
							remove_variable = nobles_leader
							set_variable = {
								name = nobles_leader
								value = prev
							}
						}
					}

					if = {
						limit = { scope:current_d_ruler.var:citizens_leader.diplomacy < this.diplomacy }

						scope:current_d_ruler = {
							remove_variable = citizens_leader
							set_variable = {
								name = citizens_leader
								value = prev
							}
						}
					}
					
				}
				var:army_leader = {
					set_variable = {
						name = is_army_leader
						value = yes
					}
				}
				var:nobles_leader = {
					set_variable = {
						name = is_nobles_leader
						value = yes
					}
				}
				var:citizens_leader = {
					set_variable = {
						name = is_citizens_leader
						value = yes
					}
				}
			}
		}
	}
}

on_dynastic_find_army_leader = {
	trigger = {
		is_ruler = yes
		has_government = dynastic_government
	}

	effect = {
		if = {
			limit = { 
				AND = {
					has_variable = army_leader
					var:army_leader = {
						is_alive = yes
						liege = root
					}
				}
			}

			every_vassal = {
				limit = { root.var:army_leader.martial < this.martial }

				root.var:army_leader = {
					remove_variable = is_army_leader
				}
				root = {
					remove_variable = army_leader
					set_variable = {
						name = army_leader
						value = prev
					}
				}
			}
			var:army_leader = {
				set_variable = {
					name = is_army_leader
					value = yes
				}
			}
		}
		else = {
			random_vassal = {
				root = {
					set_variable = {
						name = army_leader
						value = prev
					}
				}
			}
			every_vassal = {
				limit = { root.var:army_leader.martial < this.martial }

				root = {
					remove_variable = army_leader
					set_variable = {
						name = army_leader
						value = prev
					}
				}
			}
			var:army_leader = {
				set_variable = {
					name = is_army_leader
					value = yes
				}
			}
			set_variable = {
				name = army_h_amount
				value = 50
			}
		}
	}
}

on_dynastic_ruler_death = {
	trigger = {
		scope:previous_holder = {
			is_landed = yes
			has_government = dynastic_government
		}
	}
	effect = {
		set_variable = {
			name = army_h_amount
			value = scope:previous_holder.var:army_h_amount
		}
		set_variable = {
			name = nobles_h_amount
			value = scope:previous_holder.var:nobles_h_amount
		}
		set_variable = {
			name = citizens_h_amount
			value = scope:previous_holder.var:citizens_h_amount
		}
		set_variable = {
			name = army_leader
			value = scope:previous_holder.var:army_leader
		}
		set_variable = {
			name = nobles_leader
			value = scope:previous_holder.var:nobles_leader
		}
		set_variable = {
			name = citizens_leader
			value = scope:previous_holder.var:citizens_leader
		}

		scope:previous_holder = {
			remove_variable = army_h_amount
			remove_variable = nobles_h_amount
			remove_variable = citizens_h_amount
			remove_variable = army_leader
			remove_variable = nobles_leader
			remove_variable = citizens_leader
		}
	}
}

on_dynastic_ruler_title_lost = {
	trigger = {
		AND = {
			is_landed = yes
			has_government = dynastic_government
			scope:title = root.primary_title
		}
	}
	effect = {
		scope:new_holder = {
			set_variable = {
				name = army_h_amount
				value = root.var:army_h_amount
			}
			set_variable = {
				name = nobles_h_amount
				value = root.var:nobles_h_amount
			}
			set_variable = {
				name = citizens_h_amount
				value = root.var:citizens_h_amount
			}
			set_variable = {
				name = army_leader
				value = root.var:army_leader
			}
			set_variable = {
				name = nobles_leader
				value = root.var:nobles_leader
			}
			set_variable = {
				name = citizens_leader
				value = root.var:citizens_leader
			}
		}

		remove_variable = army_h_amount
		remove_variable = nobles_h_amount
		remove_variable = citizens_h_amount
		remove_variable = army_leader
		remove_variable = nobles_leader
		remove_variable = citizens_leader
	}
}
