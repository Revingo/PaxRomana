on_family_traitor = {
	trigger = {
		AND = {
			#Checks if the character already has the family_traitor modifier
			NOT = {has_character_modifier = family_traitor}
	
			#if he has not, checks if his culture has the tradition_pure_family tradition, then checks
			culture = {
				has_cultural_tradition = tradition_pure_family
			}
			#if the scope's house is the same of the scope's primary spouse's house
			NOT = { house = primary_spouse.house }
		}
	}

	#if all conditions are met, fires the familyTraitorEvent.1
	events = {
		familyTraitorEvent.1
	}
}

on_create_league = {
	trigger = {
		is_ruler = yes
		highest_held_title_tier >= 2
		government_has_flag = government_is_city-state
		NOT = {has_variable_list = league_list }
		NOT = {has_variable = in_a_league }
	}

	effect = {
		add_to_list = league_list
		set_variable = {
			name = in_a_league
			value = yes
		}
		set_variable = {
			name = is_league_leader
			value = yes
		}
	}
}

on_add_to_league = {
	trigger = {
		is_ruler = yes
		highest_held_title_tier >= 2
		government_has_flag = government_is_city-state
		NOT = {has_variable_list = league_list }
		NOT = {has_variable = in_a_league }
	}

	effect = {
		set_variable = {
			name = in_a_league
			value = root.primary_title
		}
		set_variable = {
			name = is_league_leader
			value = no
		}
	}
}

quarterly_playable_pulse = {
	on_actions = {
		on_league_funds_increase
		on_dynastic_find_army_leader
	}
}

on_league_funds_increase = {
	trigger = {
		is_ruler = yes
		has_government = city-state_government
		has_variable = funds_value
	}

	effect = {
		change_variable = {
			name = funds_value
			add = quarterly_league_funds_increase
			round = yes
		}
	}
}

on_dynastic_find_army_leader = {
	trigger = {
		is_ruler = yes
		has_government = dynastic_government
	}

	effect = {
		if = {
			limit = { has_variable = army_leader }

			every_vassal_or_below = {
				limit = { root.var:army_leader.martial < this.martial }

				if = {
					limit = { has_variable = is_army_leader }
				
					remove_variable = is_army_leader
				}
				root = {
					remove_variable = army_leader
					set_variable = {
						name = army_leader
						value = prev
					}
				}
			}
			var:army_leader = {
				set_variable = {
					name = is_army_leader
					value = true
				}
			}
		}
		else = {
			random_vassal_or_below = {
				root = {
					set_variable = {
						name = army_leader
						value = prev
					}
				}
			}
			every_vassal_or_below = {
				limit = { root.var:army_leader.martial < this.martial }

				root = {
					remove_variable = army_leader
					set_variable = {
						name = army_leader
						value = prev
					}
				}
			}
			var:army_leader = {
				set_variable = {
					name = is_army_leader
					value = true
				}
			}
		}
	}
}

on_death = {
	on_actions = {
		on_league_leader_death
		on_league_member_death
	}
}

on_league_leader_death = {
	trigger = {
		AND = {
			AND = {
				has_variable = in_a_league
				var:in_a_league = yes
			}
			NOT = { has_variable = league_owner }
		}
		
	}
	effect = {

		if = {
			limit = { has_variable_list = reintegrate_in_league }

			clear_variable_list = reintegrate_in_league
		}

		#If the character has no player heir or the player heir is landed and has a different government, then the league gets disbanded
		if = {
			limit = { 
				OR = {
					NOT = { exists = player_heir }
					AND = {
						exists = player_heir
						player_heir = {
							is_landed = yes
							NOT = { has_government = city-state_government }
						}
					}
				}
			}
			
			every_in_list = {
				variable = league_list
 

				this = {
					if = {
						limit = {has_variable = league_owner }

						remove_variable = in_a_league
						remove_variable = league_owner
						remove_character_modifier = league_funds
					}
					
					root = {
						set_variable = {
							name = member
							value = prev
						}
						every_in_list = {
							variable = league_list

							if = { 
								limit = { NOT = { root.var:member = this}}

								break_alliance = root.var:member
							}
						}
						remove_variable = member
					}
				}
			}

			remove_variable = in_a_league
			remove_character_modifier = league_funds
			remove_variable = funds_value
			clear_variable_list = league_list
		}

		#In any other case, the player heir becomes the new leader
		else = {
			every_in_list = {
				variable = league_list

				if = { 
					limit = { NOT = {root = this}}

					this = {
						remove_variable = league_owner
						set_variable = {
							name = league_owner
							value = root.player_heir
						}
					}
				}
			}

			player_heir = {
				add_to_variable_list = {
					name = league_list
					target = root.player_heir
				}
				set_variable = {
					name = in_a_league
					value = yes
				}
				add_character_modifier = {
					modifier = league_funds
				}
				set_variable = {
					name = funds_value
					value = root.var:funds_value
				}
			}

			every_in_list = {
				variable = league_list
	
				if = { 
					limit = {NOT = {this = root}}
					
					root.player_heir = {
						add_to_variable_list = {
							name = league_list
		
							target = prev
						}
						create_alliance = {
							target = prev
							allied_through_owner = root.player_heir
							allied_through_target = prev
						}
					}	
				}
			}
			remove_variable = in_a_league
			remove_character_modifier = league_funds
			remove_variable = funds_value
		}
	}
}

on_league_member_death = {
	trigger = {
		AND = {
			has_variable = in_a_league
			var:in_a_league = yes
			has_variable = league_owner
		}
		
	}
	effect = {
		if = { #If the character has no player heir or the player heir is landed and has a different government, then the city just leaves the league
			limit = {
				OR = {
					NOT = { exists = player_heir }
					AND = {
						exists = player_heir
						player_heir = {
							is_landed = yes
							NOT = { has_government = city-state_government }
						}
					}
				}
			}

			var:league_owner = {
				remove_list_variable = {
					name = league_list
	
					target = root
				}
			}
			remove_character_modifier = league_funds
			remove_variable = in_a_league
			remove_variable = league_owner

		}
		else = { #In the other cases
			if = { #If the player heir isn't in a league already, then he becomes a new league member
				limit = {
					player_heir = {
						NOT = { has_variable = in_a_league }
					}
				}

				player_heir = {
					set_variable = {
						name = in_a_league
						value = yes
					}
					set_variable = {
						name = league_owner
						value = root.var:league_owner
					}
					add_character_modifier = {
						modifier = league_funds
					}
				}
	
				var:league_owner = {
					every_in_list = {
						variable = league_list
						
						create_alliance = {
							target = root.player_heir
							allied_through_owner = this
							allied_through_target = root.player_heir
						}
					}
	
					add_to_variable_list = {
						name = league_list
						target = root.player_heir
					}
				}
			}

			#then it removes the old character from the league and removes all the variables
			var:league_owner = {
				remove_list_variable = {
					name = league_list
	
					target = root
				}
			}
			
			remove_character_modifier = league_funds
			remove_variable = in_a_league
			remove_variable = league_owner
		}
		
	}
}
