military_vassal_obligations = {
	obligation_levels = {
		default = {
			levies = {
				# If the liege has a county tier level title, the levy contribution is 40%
				value = {
					add = 0.40

					# If it has a duchy level title, it is 60%
					if = {
						limit = {
							scope:liege = {
								highest_held_title_tier > 2
							}
						}
	
						value = {
							add = 0.20
						}
					}

					# If it has a kingdom level title, it is 80%
					if = {
						limit = {
							scope:liege = {
								highest_held_title_tier > 3
							}
						}

						value = {
							add = 0.20
						}
					}

					# If it has a empire level title, it is 100%
					if = {
						limit = {
							scope:liege = {
								highest_held_title_tier = 5
							}
						}

						value = {
							add = 0.20
						}
					}
				}
			}
			tax = {
				# If the liege has a county tier level title, the military camp costs 5% of the budget
				value = {
					add = -0.05

					# If it has a duchy level title, the military camp costs 7% of the budget
					if = {
						limit = {
							scope:liege = {
								highest_held_title_tier > 2
							}
						}
	
						value = {
							add = -0.02
						}
					}

					# If it has a kingdom level title, the military camp costs 9% of the budget
					if = {
						limit = {
							scope:liege = {
								highest_held_title_tier > 3
							}
						}

						value = {
							add = -0.02
						}
					}

					# If it has a empire level title, the military camp costs 11% of the budget
					if = {
						limit = {
							scope:liege = {
								highest_held_title_tier = 5
							}
						}

						value = {
							add = -0.02
						}
					}
				}
			}
		}
	}
}
